--- FIND YOUR SUBSCRIPTION'S ALLOWED REGIONS ---

# The policy "sys.regionrestriction" controls which regions you can use.
# Get the policy assignment ID first:
az policy assignment list --query "[?name=='sys.regionrestriction'].id" -o tsv

# Then show it with the full ID (replace the ID below with the actual one):
az policy assignment show --ids "/subscriptions/YOUR_SUB_ID/providers/Microsoft.Authorization/policyAssignments/sys.regionrestriction" --query parameters

# Or get the policy definition rule directly:
az policy definition show --ids "/providers/Microsoft.Authorization/policyDefinitions/b86dabb9-b578-4d7b-b842-3b45e95769a1" --query policyRule

# Use one of those allowed regions in the deploy:
curl -sL https://raw.githubusercontent.com/sundhar-04/Reverse-Engineering-Contest/main/deploy-azure.sh | bash -s -- <REGION>
