--- VM DEPLOYED! NOW SET IT UP ---

# 1. Get the public IP
Go to portal.azure.com → Virtual machines → ReverseEngineeringVM → copy Public IP address

# 2. SSH in from Cloud Shell
ssh -i ~/Downloads/ReverseEngineeringVM_key.pem azureuser@<IP>

# 3. Install Docker
sudo apt update && sudo apt install -y docker.io docker-compose-v2
sudo usermod -aG docker $USER
# Logout and back in:
exit
ssh -i ~/Downloads/ReverseEngineeringVM_key.pem azureuser@<IP>

# 4. Clone and deploy
git clone https://github.com/sundhar-04/Reverse-Engineering-Contest.git
cd Reverse-Engineering-Contest
echo "JWT_SECRET=$(openssl rand -hex 64)" > .env
sudo docker compose up -d

# 5. Open ports 3000 and 8000
Go to portal → ReverseEngineeringVM → Networking → Add inbound port rule:
  - Port 3000, Priority 1010, Name: frontend
  - Port 8000, Priority 1020, Name: backend

# 6. Access
http://<IP>:3000
Admin: http://<IP>:3000/login → admin / weguysfromcodingninjas
